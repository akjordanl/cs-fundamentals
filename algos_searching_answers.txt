function binarySearch (dictionary, word) {
  var lo = 0;
  var hi = dictionary.length - 1;

  while ( lo <= hi ) {
    var mid = Math.floor( (lo + hi ) / 2 );
    if ( dictionary[mid] === word ) {
      return word + " is at index " + mid;
    } else if (dictionary[mid] > word) {
      hi = mid;
    } else if (dictionary[mid] < word) {
      lo = mid;
    }
  }
}
